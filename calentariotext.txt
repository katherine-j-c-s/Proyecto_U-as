const calendarioGrid = document.querySelector('.calendario-grid');
    const hoy = new Date();
    let firstDayOfMonth = new Date(hoy.getFullYear(), hoy.getMonth(), 1);
    const lastDayOfMonth = new Date(hoy.getFullYear(), hoy.getMonth() + 1, 0);

    const daysInWeek = 7;
    const totalDaysToShow = 35; // 7x5 grid

    // Rellenar los días en la cuadrícula del calendario
    let currentDate = hoy.getDate();
    const daysLeftInMonth = lastDayOfMonth.getDate() - currentDate;
    
    for (let day = 1; day <= lastDayOfMonth.getDate(); day++) {
        const dayDiv = document.createElement('div');
        dayDiv.textContent = day;
        if (day < currentDate) {
            dayDiv.classList.add('desactivo');
        } else if (day === currentDate) {
            dayDiv.classList.add('activo');
        }
        calendarioGrid.appendChild(dayDiv);
    }

    // Si faltan menos de 7 días, añadir los primeros días del siguiente mes
    if (daysLeftInMonth < 7) {
        let nextMonthDay = 1;
        while (calendarioGrid.children.length < totalDaysToShow) {
            const dayDiv = document.createElement('div');
            dayDiv.textContent = nextMonthDay;
            calendarioGrid.appendChild(dayDiv);
            nextMonthDay++;
        }
    } else {
        // Si no faltan menos de 7 días, completamos hasta 35 días con los días del mes actual
        for (let day = 1; calendarioGrid.children.length < totalDaysToShow; day++) {
            const dayDiv = document.createElement('div');
            if (lastDayOfMonth.getDate() < day) {
                dayDiv.textContent = firstDayOfMonth;
                firstDayOfMonth++;
            }else{
                dayDiv.textContent = currentDate + daysLeftInMonth + day;
            }
            calendarioGrid.appendChild(dayDiv);
        }
    }
